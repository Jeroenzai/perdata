<?php
//Opvragen file-name van huigie bestand
$file = "content.txt";
$this_file = basename(__FILE__);

//Controleer of data is gepost en of 'new_contents' hier onderdeel van is.
if ( $_POST && array_key_exists('new_contents', $_POST) ) {

    //Schrijf de waarde van 'new_contents' naar het bestand
    file_put_contents($file, $_POST['new_contents']);
    
    //Redirect naar eigen bestand (herladen)
    header( "Location: http://" .$_SERVER['HTTP_HOST'] .dirname($_SERVER['PHP_SELF']) . "/" . $this_file );
     
    //Exit PHP script.
    exit;
}
?>

<!--Het HTML formulier. Input in de textarea wordt als POST data naar dit bestand gezonden bij een submit.›-->
<!DOCTYPE html>
<html>
  <head>
    <title>UU Wiki</title>
  </head>
  <body>
    <form method="POST">
      <textarea name="new_contents" rows="70" cols="70"><?=
         htmlspecialchars( file_get_contents($file) );
      ?></textarea>
      <input type="submit" value="Submit"/>
    </form>
  </body>
</html>
